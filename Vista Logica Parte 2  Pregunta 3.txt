Create View Buses

AS

select distinct Datepart(hour, fecha_transaccion) as hora, bus
from mibus.dbo.transacciones
group by Datepart(hour, fecha_transaccion), bus


//---

select hora, count(bus) as cantidad
from buses
group by hora
order by 1,2
